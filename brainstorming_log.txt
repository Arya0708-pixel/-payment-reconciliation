5. Step 4 – Reconciliation Engine (reconciliation.py)
import pandas as pd

def reconcile(platform, bank):
    report = {}

    report["total_platform"] = len(platform)
    report["total_bank"] = len(bank)

    # Match by ID
    matched = platform[platform["id"].isin(bank["id"])]
    report["matched"] = len(matched)

    # Late settlements
    merged = platform.merge(bank, on="id", how="inner")
    merged["date"] = pd.to_datetime(merged["date"])
    merged["settlement_date"] = pd.to_datetime(merged["settlement_date"])

    late = merged[
        merged["settlement_date"].dt.month != merged["date"].dt.month
    ]
    report["late_settlements"] = len(late)

    # Duplicates
    report["bank_duplicates"] = bank.duplicated(subset=["id"]).sum()

    # Orphan refunds
    refunds = platform[platform["type"] == "refund"]
    orphan_refunds = refunds[~refunds["id"].isin(bank["id"])]
    report["orphan_refunds"] = len(orphan_refunds)

    # Rounding difference
    report["platform_total"] = platform["amount"].sum()
    report["bank_total"] = bank["amount"].sum()
    report["difference"] = report["platform_total"] - report["bank_total"]

    return report

def reconcile_without_ids(platform, bank):
    matches = []

    for _, p in platform.iterrows():
        for _, b in bank.iterrows():
            if abs(p["amount"] - b["amount"]) < 0.01:
                matches.append(p["id"])

    return len(matches)

This shows:
